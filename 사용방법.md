# hangulDisp 사용방법

이 문서는 통합 어댑터 기반의 간단한 예제를 제공합니다.
각 예제는 약 20~30라인으로 구성되어 있습니다.

## 공통 준비

- hangulDisp 라이브러리
- 필요한 디스플레이 라이브러리 (u8glib/u8g2/gxepd2/tftlcd)
- 한글 폰트 헤더 (예: H01_Font.h)

## u8glib 예제 (OLED)

```cpp
#include <U8glib.h>
#include "hangulDisp.h"
#include "fonts/hangul/H01_Font.h"

U8GLIB_SSD1306_128X64 u8g(U8G_I2C_OPT_NONE);

void drawPixel(int16_t x, int16_t y, uint16_t color) {
    u8g.setColorIndex(color ? 1 : 0);
    u8g.drawPixel(x, y);
}

hangulDisp<Font_H01> hangul(drawPixel);

void setup() {
    u8g.setFont(u8g_font_6x10);
}

void loop() {
    u8g.firstPage();
    do {
        hangul.setCursor(0, 20);
        hangul.setTextColor(HG_COLOR_WHITE);
        hangul.print("Hello Hangul");
    } while (u8g.nextPage());
}
```

## u8g2 예제 (OLED)

```cpp
#include <U8g2lib.h>
#include "hangulDisp.h"
#include "fonts/hangul/H01_Font.h"

U8G2_SSD1306_128X64_NONAME_F_HW_I2C u8g2(U8G2_R0, U8X8_PIN_NONE);

void drawPixel(int16_t x, int16_t y, uint16_t color) {
    u8g2.setDrawColor(color ? 1 : 0);
    u8g2.drawPixel(x, y);
}

hangulDisp<Font_H01> hangul(drawPixel);

void setup() {
    u8g2.begin();
    u8g2.clearBuffer();
    hangul.setCursor(0, 20);
    hangul.print("Hello Hangul");
    u8g2.sendBuffer();
}

void loop() {}
```

## gxepd2 예제 (E-Paper)

```cpp
#include <GxEPD2_BW.h>
#include "hangulDisp.h"
#include "fonts/hangul/H01_Font.h"

GxEPD2_BW<GxEPD2_154, GxEPD2_154::HEIGHT> display(
    GxEPD2_154(SS, 8, 9, 7)
);

void drawPixel(int16_t x, int16_t y, uint16_t color) {
    display.drawPixel(x, y, color ? GxEPD_WHITE : GxEPD_BLACK);
}

hangulDisp<Font_H01> hangul(drawPixel);

void setup() {
    display.init();
    display.setFullWindow();
    display.firstPage();
    do {
        display.fillScreen(GxEPD_WHITE);
        hangul.setCursor(10, 30);
        hangul.print("Hello Hangul");
    } while (display.nextPage());
}

void loop() {}
```

## tftlcd 예제 (TFT_eSPI)

```cpp
#include <TFT_eSPI.h>
#include "hangulDisp.h"
#include "fonts/hangul/H01_Font.h"

TFT_eSPI tft = TFT_eSPI();

void drawPixel(int16_t x, int16_t y, uint16_t color) {
    tft.drawPixel(x, y, color ? TFT_WHITE : TFT_BLACK);
}

hangulDisp<Font_H01> hangul(drawPixel);

void setup() {
    tft.init();
    tft.fillScreen(TFT_BLACK);
    hangul.setCursor(10, 30);
    hangul.setTextColor(HG_COLOR_WHITE);
    hangul.print("Hello Hangul");
}

void loop() {}
```
